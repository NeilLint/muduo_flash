/*  TODO 向量点积加速执行
    dot: 向量点积
    *y += x1[0]*x2[0] + x1[1]*x2[1] + ... + x1[dim-1]*x2[dim-1]
*/
void CBackend::dot(float *y, float *x1, float *x2, int dim) {
    if (dim <= 0 || x1 == nullptr || x2 == nullptr || y == nullptr) return;

    size_t data_size_bytes = dim * sizeof(float);

    // 1. 分配设备内存
    float *d_x1 = nullptr, *d_x2 = nullptr, *d_y = nullptr;
    HIP_CHECK(hipMalloc(&d_x1, data_size_bytes));
    HIP_CHECK(hipMalloc(&d_x2, data_size_bytes));
    HIP_CHECK(hipMalloc(&d_y, sizeof(float)));  // 结果 y 是标量，所以只分配一个 float

    // 2. 将 CPU 数据拷贝到 GPU
    HIP_CHECK(hipMemcpy(d_x1, x1, data_size_bytes, hipMemcpyHostToDevice));
    HIP_CHECK(hipMemcpy(d_x2, x2, data_size_bytes, hipMemcpyHostToDevice));

    // 3. 调用 hipblas 函数计算 dot 产品
    HIPBLAS_CHECK(hipblasSdot(blas_handle, 
                              dim,    // 向量维度
                              d_x1,   // 向量 x1 的设备指针
                              1,      // 向量 x1 的增量
                              d_x2,   // 向量 x2 的设备指针
                              1,      // 向量 x2 的增量
                              d_y));  // 结果存储到设备内存中的 y

    // 4. 将计算结果从 GPU 内存拷贝回 CPU
    HIP_CHECK(hipMemcpy(y, d_y, sizeof(float), hipMemcpyDeviceToHost));

    // 5. 释放设备内存
    HIP_CHECK(hipFree(d_x1));
    HIP_CHECK(hipFree(d_x2));
    HIP_CHECK(hipFree(d_y));
}
